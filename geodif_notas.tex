\section{TeorĆ­a de categorĆ­as}

\begin{defn}
Una \emph{categorĆ­a} $\sfC$ consiste en
\begin{itemize}
  \item una clase $\mathrm{Obj}(\sfC)$, cuyos elementos se llaman los \emph{objetos} de $\sfC$,
\item para cada par de objetos $A$ y $B$ de $\sfC$, una clase $\Hom_{\sfC}(A,B)=\Hom(A,B)$, cuyos elementos $f$ se denominan \emph{morfismos de $A$ a $B$} y se denotan como flechas $f:A\rightarrow B$,
\end{itemize}
y una \emph{ley de composiciĆ³n}, para cada terna $(A,B,C)$ de objetos de $\sfC$
\begin{equation*}
\circ_{\sfC}=\circ: \Hom(A,B) \times \Hom(B,C) \rightarrow \Hom(A,C), \ (f,g)\mapsto g\circ f
\end{equation*}
que cumple la propiedad asociativa y tal que, para cada objeto $A$ de $\sfC$, existe un morfismo $\id_A:A\rightarrow A$ que actĆŗa como la unidad para la operaciĆ³n de composiciĆ³n.

Si, para cada par de objetos $(A,B)$ de $\sfC$, la clase $\Hom(A,B)$ es un conjunto, entonces decimos que $\sfC$ es una categorĆ­a \emph{localmente pequeĆ±a}. Si, ademĆ”s, la clase $\mathrm{Obj}(\sfC)$ es un conjunto, decimos que $\sfC$ es una categorĆ­a \emph{pequeĆ±a}.
\end{defn}

\begin{defn}
Un \emph{isomorfismo} en una categorĆ­a $\sfC$ es un morfismo $f:A\rightarrow B$ tal que existe otro morfismo $f^{-1}:B\rightarrow A$ de modo que $f^{-1}\circ f = \id_A$ y $f\circ f^{-1}= \id_B$.
\end{defn}

\begin{defn}
Un \emph{funtor} (o \emph{funtor covariante}) $F:\sfC\rightarrow \sfD$ entre dos categorĆ­as $\sfC$ y $\sfD$, asigna, a cada objeto $A$ de $\sfC$, un objeto $F(A)$ de $\sfD$ y, a cada morfismo $f:A\rightarrow B$ en $\sfC$, un morfismo $F(f):F(A)\rightarrow F(B)$, de forma que $F(f\circ g) = F(f) \circ F(g)$.
\end{defn}

\subsection{} Dada una categorĆ­a $\sfC$, podemos definir su \emph{categorĆ­a opuesta} $\sfC^{\op}$ como la categorĆ­a cuyos objetos son los objetos de $\sfC$ y cuyos morfismos cumplen
\begin{equation*}
\Hom_{\sfC^{\op}}(A,B) = \Hom_{\sfC}(B,A)
\end{equation*}
y de modo que $g \circ_{\sfC^{\op}} f = f\circ_{\sfC} g$.
Un \emph{funtor contravariante} de $\sfC$ a $\sfD$ se define como un funtor $F:\sfC^{\op}\rightarrow \sfD$.

\begin{prop}
Dado un funtor $F:\sfC\rightarrow \sfD$, para todo isomorfismo $f:A\rightarrow B$ en $\sfC$, el morfismo $F(f):F(A)\rightarrow F(B)$ es un isomorfismo en $\sfD$, con $F(f)^{-1}=f^{-1}$.
\end{prop}

\subsection{} Es claro que los funtores se pueden componer, y que en cada categorĆ­a $\sfC$ se puede definir su funtor identidad $\id_{\sfC}:\sfC\rightarrow \sfC$, que envĆ­a cada objeto y cada morfismo en sĆ­ mismo. Esto permitirĆ­a a priori definir una categorĆ­a $\mathsf{Cat}$ cuyos objetos son las categorĆ­as y cuyos morfismos son los funtores entre ellas. En principio esto puede dar lugar a paradojas lĆ³gicas de tipo Russell (Āæes $\mathsf{Cat}$ un objeto de sĆ­ misma?), que se pueden evitar limitando el tamaĆ±o de las mismas. Por ejemplo, podemos considerar la categorĆ­a $\mathsf{CatLP}$ de las categorĆ­as localmente pequeĆ±as, que a su vez no serĆ” localmente pequeĆ±a. En cualquier caso, existe una nociĆ³n natural de ``isomorfismo de categorĆ­as'', aunque normalmente nos interesarĆ” una condiciĆ³n menos restrictiva, la nociĆ³n de ``equivalencia''. Para definirla, vamos a introducir una nociĆ³n de ``morfismos de funtores''.

\begin{defn}
  Sean $F,G:\sfC\rightarrow \sfD$ dos funtores. Una \emph{transformaciĆ³n natural} o \emph{morfismo de funtores} $\alpha:F\rightarrow G$ asigna, a cada objeto $A$ de $\sfC$, un morfismo \[\alpha_A:F(A)\rightarrow G(A)\] de modo que, para cada morfismo $f:A\rightarrow B$ en $\sfC$, el siguiente diagrama conmuta
  \begin{center}
    \begin{tikzcd}
     F(A) \ar{r}{F(f)} \ar{d}{\alpha_A} & F(B) \ar{d}{\alpha_B} \\
     G(A) \ar{r}{G(f)} & G(B).
    \end{tikzcd}
  \end{center}
Si $\alpha_A$ es un isomorfismo para todo objeto $A$ de $\sfC$, entonces se dice que $\alpha$ es un \emph{isomorfismo de funtores}.
\end{defn}

\begin{defn}
Una \emph{equivalencia de categorĆ­as} es un par de funtores $F:\sfC \rightarrow \sfD$ y $G:\sfD \rightarrow \sfC$ de forma que $G\circ F$ es isomorfo a $\id_{\sfC}$ y $F\circ G$ es isomorfo a $\id_{\sfD}$.

Un funtor $F:\sfC\rightarrow \sfD$ entre categorĆ­as localmente pequeĆ±as se dice \emph{plenamente fiel} si, para cada par de objetos $(A,B)$ de $\sfC$, se tiene que la aplicaciĆ³n
\begin{equation*}
F: \Hom(A,B) \rightarrow \Hom(F(A),F(B))
\end{equation*}
es biyectiva.

Un funtor $F:\sfC \rightarrow \sfD$ se dice \emph{esencialmente sobreyectivo} si, para cada objeto $X$ de $\sfD$, existe un objeto $A$ de $\sfC$ tal que $F(A)$ es isomorfo a $X$.
\end{defn}

\begin{prop}
Si $(F,G)$ determina una equivalencia de categorĆ­as, entonces los funtores $F$ y $G$ son plenamente fieles y esencialmente sobreyectivos. RecĆ­procamente, si $F$ es un funtor plenamente fiel y esencialmente sobreyectivo, entonces existe otro funtor $G=F^{-1}$ tal que $(F,F^{-1})$ es una equivalencia de categorĆ­as.
\end{prop}

Otra forma de definir equivalencias de categorĆ­as es en tĆ©rminos de adjunciones.

\begin{defn}
  Una \emph{adjunciĆ³n} entre dos categorĆ­as $\sfC$ y $\sfD$ es un par de funtores $F:\sfC\rightarrow \sfD$ y $G:\sfD \rightarrow \sfC$ junto con dos transformaciones naturales $\eta:\id_{\sfC} \rightarrow G\circ F$ y $\varepsilon:F\circ G \rightarrow \id_{\sfD}$, llamadas la \emph{unidad} y la \emph{counidad}, respectivamente, de modo que los siguientes diagramas conmutan
  \begin{center}
    \begin{tikzcd}
      G \ar{rd}{\id_G} \rar{\eta \circ G} & G\circ F \circ G \dar{G \circ \varepsilon} \\
      & G,
    \end{tikzcd}
    \hspace{1cm}
    \begin{tikzcd}
      F \ar{rd}{\id_F} \rar{F\circ \eta} & F \circ G \circ F \dar{\varepsilon \circ F} \\
      & F.
    \end{tikzcd}
  \end{center}
\end{defn}

\subsection{} NĆ³tese que una equivalencia de categorĆ­as es una adjunciĆ³n en la que la unidad y la counidad son isomorfismos. Por otra parte, dada una adjunciĆ³n $(F,G)$ podemos definir, para cada par de objetos $A$ en $\sfC$ y $X$ en $\sfD$, una biyecciĆ³n
\begin{equation*}
\varphi_{A,X}:\Hom_{\sfD}(F(A),X) \rightarrow \Hom_{\sfC}(A,G(X)),\ f \mapsto G(f) \circ \eta_A,
\end{equation*}
que es natural en el sentido de que, si tomamos otro par $(B,Y)$, con morfismos $f:B\rightarrow A$ y $g:X\rightarrow Y$, entonces el siguiente diagrama conmuta
\begin{center}
  \begin{tikzcd}
   \Hom_{\sfD}(F(A),X) \ar{rr}{g\circ - \circ F(f)} \ar{d}{\varphi_{A,X}} && \Hom_{\sfD}(F(B),Y)  \ar{d}{\varphi_{B,Y}} \\
   \Hom_{\sfC}(A,G(X)) \ar{rr}{G(g)\circ - \circ f} && \Hom_{\sfC}(B,G(Y)).
  \end{tikzcd}
\end{center}

\begin{defn}
Sean $\sfC$ una categorĆ­a cualquiera y $\Set$ la categorĆ­a de los conjuntos. Un \emph{prehaz} en $\sfC$ es un funtor $F:\sfC^{\op}\rightarrow \Set$. Denotamos por $\PSh(\sfC)$ la categorĆ­a de los prehaces en $\sfC$.

\subsection{} Recordemos que un \emph{espacio afĆ­n} sobre un espacio vectorial real $V$ es un conjunto $A$ junto con una acciĆ³n libre y transitiva por la derecha de $V$. (Decimos que $A$ es un $V$-torsor). Un \emph{espacio euclidiano} es un espacio afĆ­n $A$ sobre $V$ equipado con un producto escalar
\begin{align*}
  \langle-,-\rangle: V\times V & \longrightarrow \mathbb{R} ,
\end{align*}
que recordamos que es una forma bilineal simĆ©trica y definida positiva. Recordemos tambiĆ©n que una base $\left\{ e_1,\dots,e_n \right\}$ de $V$ se denomina \emph{ortonormal} si $\langle e_i, e_j \rangle = \delta_{ij}$. El algoritmo de Gram--Schmidt permite producir una base ortonormal a partir de una base cualquiera.

\begin{defn}
  Una \emph{mĆ©trica riemanniana} (o, simplemente, una \emph{mĆ©trica}) en una variedad diferenciable $M$ es una elecciĆ³n de producto escalar $\langle-,-\rangle_x:T_xM \times T_xM \rightarrow \RR$ en cada espacio tangente, que depende del punto $x$ de forma diferenciable. Equivalentemente, una mĆ©trica riemanniana es un tensor simĆ©trico de tipo $(0,2)$, es decir $g\in \mathcal{S}^0_2(M)$.

  Un par $(M,g)$ formado por una variedad diferenciable con una mĆ©trica riemanniana en ella, se denomina una \emph{variedad riemanniana}. Una \emph{isometrĆ­a} entre dos variedades Riemannianas $(M_1,g_1)$ y $(M_2,g_2)$ es una aplicaciĆ³n diferenciable  $f:M\rightarrow N$ tal que $f^*g_2 = g_1$.
\end{defn}

\subsection{Existencia} La existencia de mĆ©tricas Riemannianas en cualquier variedad diferenciable es clara. Basta tomar, para cada sistema de coordenadas local $\bm{x}:U\rightarrow \RR^n$, el pullback del producto escalar estĆ”ndar en $\RR^n$. Es decir, tomamos la mĆ©trica Riemanniana $g_U$ en $U$ tal que la base $\left\{ \partial_i:i=1,\dots,n \right\}$ es ortonormal. Para recuperar una mĆ©trica definida globalmente, basta tomar una particiĆ³n diferenciable de la unidad $\rho_U$ subordinada a la familia de abiertos $U$ y definir
\begin{equation*}
g = \sum_{U} \rho_U g_U.
\end{equation*}

\subsection{} Tomando coordenadas locales $x^1,\dots,x^n$ en $U\subset M$, podemos escribir una mĆ©trica como
\begin{equation*}
g = g_{ij} dx^i dx^j,
\end{equation*}
donde $g_{ij}:U\rightarrow \RR$ cumple
\begin{equation*}
g_{ij}(x) = \langle \partial_i|_x, \partial_j|_x \rangle_x.
\end{equation*}

\subsection{} NĆ³tese que una mĆ©trica riemanniana $g$ en una $n$-variedad orientada $M$ tiene asociada de forma natural una forma de volumen $d \mathrm{vol}_g \in \Omega^n(M)$ definida como sigue. Para cada $p\in M$, se toma una base ortonormal positiva $e_1,\dots,e_n$ del espacio tangente $TM$, con base dual $\theta^1,\dots,\theta^n$, y se toma $d\mathrm{vol}_g=\theta^1\wedge \cdots \wedge \theta^n$. Esta forma estĆ” bien definida ya que, si se toma otra base ortonormal positiva $e_1',\dots,e_n'$, Ć©sta difiere de $e_1,\dots,e_n$ por una matriz $A \in \mathrm{SO}_n(\RR)$, de modo que \[(\theta')^1\wedge \cdots \wedge (\theta')^n=\det(A) \theta^1\wedge \cdots \wedge \theta^n = \theta^1\wedge \cdots \wedge \theta^n.\]

Si $x^1,\dots,x^n$ son coordenadas locales, entonces podemos escribir
\begin{equation*}

\end{equation*}

\subsection{El programa de Erlangen}
Con motivo de su ingreso en la Universidad de Erlangen (Alemania), Felix Klein publicĆ³ su famoso ``\textit{Erlanger Programm}'', uno de los trabajos mĆ”s influyentes en la geometrĆ­a de todos los tiempos. La idea fundamental que propuso Klein es que entender una estructura geomĆ©trica esencialmente corresponde a entender su \emph{grupo de transformaciones}. Esta idea ha trascendido hasta nuestros tiempos, y vamos a explicar cĆ³mo se manifiesta en las geometrĆ­as que ya hemos explorado: la geometrĆ­a lineal, la geometrĆ­a afĆ­n, la geometrĆ­a euclidiana y la geometrĆ­a riemanniana.

\subsection{GeometrĆ­a lineal} El grupo de automorfismos de un espacio vectorial $V$ es el grupo general lineal \[\mathrm{GL}(V)=\mathrm{Aut}(V) = \left\{ f:V\rightarrow V \text{ lineal e invertible} \right\}.\]
Si $V=\RR^n$, entonces $\mathrm{GL}(V)=\mathrm{GL}_n(\RR)$ es el grupo general lineal formado por las matrices $n\times n$ invertibles. En particular, $\mathrm{GL}_1(\RR)=\RR^\times$.

Si $V$ tiene dimensiĆ³n $n$, el conjunto $\cB(V)$ de sus bases se identifica con
\begin{equation*}
\cB(V) = \left\{b: \RR^n \overset{\sim}{\rightarrow} V \text{ isomorfismo lineal}\right\}.
\end{equation*}
Si $V=\RR^n$, entonces simplemente $\cB(\RR^n)=\GL_n(\RR)$, pero en general $\cB(V)$ es un torsor para el grupo $\GL_n(\RR)$. Es decir, existe una acciĆ³n libre y transitiva por la derecha de $\GL_n(\RR)$ en $\cB(V)$. Esta acciĆ³n es la acciĆ³n de ``cambio de base'': dada $g:\RR^n\rightarrow \RR^n$ lineal e invertible, definimos
\begin{equation*}
b\cdot g = b\circ g.
\end{equation*}

\subsection{GeometrĆ­a afĆ­n} Un espacio afĆ­n $A$ sobre un espacio vectorial $V$ es un $V$-torsor, de modo que dados dos puntos $a,b \in A$, podemos definir el ``vector que los une'' $a-b \in V$. Una aplicaciĆ³n afĆ­n $f:A\rightarrow A$ de un espacio afĆ­n en sĆ­ mismo es una aplicaciĆ³n tal que existe una aplicaciĆ³n lineal $g:V\rightarrow V$ que cumple
\begin{equation*}
f(a)-f(b) = g(a-b),
\end{equation*}
para cualesquiera dos puntos $a,b\in A$. Denotamos por $\mathrm{Aff}(A)$ el grupo de las aplicaciones afines invertibles. La asignaciĆ³n $f\mapsto g$ define una aplicaciĆ³n $\mathrm{Aff}(A)\rightarrow \mathrm{GL}(V)$ que exhibe $\mathrm{Aff}(A)$ como una extensiĆ³n
\begin{center}
  \begin{tikzcd}
0 \rar & V \rar & \mathrm{Aff}(A) \rar & \mathrm{GL}(V) \rar & 1,
  \end{tikzcd}
\end{center}
donde la aplicaciĆ³n $V\rightarrow \mathrm{Aff}(A)$ estĆ” determinada por la acciĆ³n de traslaciĆ³n de $V$ en $A$. Esta extensiĆ³n escinde al elegir un punto base $a_0$ de $A$. En ese caso, podemos escribir todo punto $a$ de $A$ como $a=a_0+v$ para un Ćŗnico vector $v\in V$ y, para toda aplicaciĆ³n lineal $g:V\rightarrow V$, la aplicaciĆ³n $f(a+v)=a+g(v)$ es una aplicaciĆ³n afĆ­n.

Una referencia afĆ­n estĆ” dada por un par $(a,b)$, donde $a\in A$ es un punto y $b:\RR^n\rightarrow V$ es una base; equivalentemente, una referencia estĆ” dada por un isomorfismo de espacios afines $\mathbb{A}^n \rightarrow A$, donde $\mathbb{A}^n$ es el espacio afĆ­n estĆ”ndar. De igual forma, el conjunto $\cB(A)$ de las referencias afines de $A$ es un torsor para el grupo $\mathrm{Aff}_n(\RR)=\mathrm{Aff}(\mathbb{A}^n)$ de automorfismos afines del espacio afĆ­n estĆ”ndar.

\subsection{GeometrĆ­a euclidiana} Si $(V,\langle-,-\rangle)$ es un espacio vectorial con un producto escalar, se define el correspondiente grupo ortogonal $\mathrm{O}(V,\langle-,-\rangle)\subset \mathrm{GL}(V)$ como el subgrupo de los automorfismos lineales de $V$ que preservan el producto escalar. Es decir,
\begin{equation*}
\mathrm{O}(V,\langle-,-\rangle) = \left\{ f:V\rightarrow V: \langle fv,fw\rangle = \langle v,w \rangle, \  \forall v,w \in V \right\}.
\end{equation*}
Si $(V,\langle-,-\rangle)$ es $\RR^n$ con el producto escalar estĆ”ndar, entonces su grupo ortogonal es el subgrupo $\mathrm{O}_n(\RR)\subset \mathrm{GL}_n(\RR)$ de las matrices ortogonales, que son aquellas $A$ que satisfacen
\begin{equation*}
A A^T = \mathbb{I}.
\end{equation*}

NĆ³tese que $\mathrm{O}_1(\RR)$ se identifica con el grupo de dos elementos $\left\{ \pm 1 \right\}$, mientras que $\mathrm{O}_n(\RR)$ tiene en general dos componentes conexas. La componente conexa neutra, denotada por $\mathrm{SO}_n(\RR)$, contiene a las matrices ortogonales con determinante $1$, que son precisamente aquellas que determinan las \emph{rotaciones} del espacio $n$-dimensional. La otra componente conexa contiene a las composiciones de las rotaciones con las \emph{reflexiones}, que tienen determinante $-1$.

MĆ”s generalmente, los automorfismos $\mathrm{Euc}(E)$ de un espacio euclidiano $E$ son aquellos automorfismos afines que preservan el producto escalar en cada punto. De igual manera, tenemos una extensiĆ³n
\begin{center}
  \begin{tikzcd}
0 \rar & V \rar & \mathrm{Euc}(E) \rar & \mathrm{O}(V,\langle-,-\rangle) \rar & 1.
  \end{tikzcd}
\end{center}
El grupo $\mathrm{Euc}_n(\RR)=\mathrm{Euc}(\RR^n)$ del espacio euclidiano estĆ”ndar se denomina el grupo euclidiano.

\subsection{GeometrĆ­a riemanniana} Sea $M$ una variedad diferenciable. El papel del torsor de bases lo juega ahora el llamado \emph{fibrado de referencias}
\begin{equation*}
\cB(M) = \left\{ (x,b): x\in M, b:\RR^n \overset{\sim}{\rightarrow} T_x M \text{ isomorfismo lineal} \right\},
\end{equation*}
que estĆ” dotado de una acciĆ³n de $\mathrm{GL}_n(\RR)$ por cambio de base y de una proyecciĆ³n natural $\pi:\cB(M)\rightarrow M$, $(x,b)\mapsto x$. Esta proyecciĆ³n determina un espacio fibrado sobre $M$ cuyas fibras son torsores para el grupo $\mathrm{GL}_n(\RR)$. Este es el primer ejemplo de un \emph{fibrado principal}. Un sistema de coordenadas $\bm{x}:U\rightarrow \RR^n$ determina una secciĆ³n $U\rightarrow \cB(M)$ dada por
\begin{equation*}
(x^1,\dots,x^n) \mapsto \left( (x^1,\dots,x^n), (\partial_1,\dots,\partial_n) \right).
\end{equation*}

Si ademĆ”s $M$ estĆ” equipada con una mĆ©trica riemanniana, podemos considerar el \emph{fibrado de referencias ortonormales}
\begin{equation*}
\cB_{\mathrm{O}}(M) = \left\{ (x,b): x\in M, b:\RR^n \overset{\sim}{\rightarrow} T_x M \text{ isometrĆ­a} \right\}.
\end{equation*}
De nuevo, la proyecciĆ³n natural a $M$ determina un fibrado cuyas fibras son torsores para $\mathrm{O}_n(\RR)$. Sin embargo, no cabe esperar a priori que unas coordenadas locales vayan a determinar una secciĆ³n de este fibrado, ya que, como veremos, no toda variedad riemanniana es localmente isomĆ©trica al espacio euclidiano estĆ”ndar. Veremos que, de hecho, existe una obstrucciĆ³n natural a la existencia de esta isometrĆ­a: la curvatura.

\section{Curvatura intrĆ­nseca}

\begin{prop}[ParametrizaciĆ³n por el arco]
Todas las $1$-variedades Riemannianas son localmente isomĆ©tricas. Concretamente: si $(C,g)$ es una $1$-variedad Riemanniana difeomorfa a $\RR$, entonces $C$ es isomĆ©trica al espacio euclidiano $\RR$ con su producto escalar estĆ”ndar.
\end{prop}

\begin{proof}
  En efecto, la prueba consiste en parametrizar por el arco. Un difeomorfismo $C\rightarrow \RR$ define una coordenada global $x$ en $C$. Definimos $g(x)=\langle \partial_x,\partial_x\rangle_x$ y buscamos una nueva coordenada $y$ tal que $h(y)=\langle \partial_y,\partial_y\rangle_y=1$. Por tanto, tenemos
  \begin{equation*}
g = h \frac{\partial y}{\partial x} \frac{\partial y}{\partial x} = \left( \frac{\partial y}{\partial x} \right)^2.
  \end{equation*}
  Por tanto, podemos fijar un punto $x_0 \in C$ y definir
  \begin{equation*}
y(x) = \int_{x_0}^x \sqrt{g(t)} dt,
  \end{equation*}
que satisface la ecuaciĆ³n anterior y por tanto es una isometrĆ­a.
\end{proof}

\subsection{} Una pregunta natural es si el resultado anterior generaliza a dimensiĆ³n superior. Veremos que esto no es cierto, ya que existe un invariante local, llamado la \emph{curvatura}, que debe ser preservado por isometrĆ­as, y que no siempre ha de anularse.

En efecto, para generalizar el resultado anterior, queremos hallar un sistema de coordenadas $\bm{y}$, con $h_{ab}=\langle \partial_{y^a},\partial_{y^b} \rangle = \delta_{ab}$ tal que
\begin{equation*}
g_{ij} = h_{ab} \frac{\partial y^a}{\partial x^i} \frac{\partial y^b}{\partial x^j} = \sum_a \frac{\partial y^a}{\partial x^i} \frac{\partial y^a}{\partial x^j}.
\end{equation*}
Nos encontramos ahora con un sistema de ecuaciones en derivadas parciales, y nos preguntamos por la existencia de soluciones. Veremos que existe una obstrucciĆ³n a esta existencia.

Comenzamos derivando
\begin{equation*}
\frac{\partial g_{ij}}{\partial x^k} = \sum_a \frac{\partial^2 y^a}{\partial x^k \partial x^i} \frac{\partial y^a}{\partial x^j} +  \frac{\partial y^a}{\partial x^i}\frac{\partial^2 y^a}{\partial x^k \partial x^j}.
\end{equation*}
Permutando Ć­ndices, obtenemos
\begin{equation*}
\frac{\partial g_{ik}}{\partial x^j} = \sum_a \frac{\partial^2 y^a}{\partial x^j \partial x^i} \frac{\partial y^a}{\partial x^k} +  \frac{\partial y^a}{\partial x^i}\frac{\partial^2 y^a}{\partial x^j \partial x^k},
\end{equation*}
\begin{equation*}
\frac{\partial g_{jk}}{\partial x^i} = \sum_a \frac{\partial^2 y^a}{\partial x^i \partial x^j} \frac{\partial y^a}{\partial x^k} +  \frac{\partial y^a}{\partial x^j}\frac{\partial^2 y^a}{\partial x^i \partial x^k}.
\end{equation*}
Sumamos las dos primeras y restamos la tercera para obtener
\begin{equation*}
\frac{1}{2} \left( \frac{\partial g_{ij}}{\partial x^k} + \frac{\partial g_{ik}}{\partial x^j} - \frac{\partial g_{jk}}{\partial x^i} \right) = \sum_a \frac{\partial y^a}{\partial x^i}\frac{\partial^2 y^a}{\partial x^j \partial x^k}.
\end{equation*}
Multiplicamos ahora por $\frac{\partial y^b}{\partial x^l} g^{li}$ y concluimos
\begin{equation*}
\frac{\partial y^b}{\partial x^l} \left[ \frac{g^{li}}{2} \left( \frac{\partial g_{ij}}{\partial x^k} + \frac{\partial g_{ik}}{\partial x^j} - \frac{\partial g_{jk}}{\partial x^i} \right)  \right] = \sum_a \frac{\partial^2 y^a}{\partial x^j \partial x^k}g^{li}\frac{\partial y^a}{\partial x^i}\frac{\partial y^b}{\partial x^l} =\frac{\partial^2 y^b}{\partial x^j \partial x^k} ,
\end{equation*}
donde hemos usado que $g^{li} \frac{\partial y^a}{\partial x^i}\frac{\partial y^b}{\partial x^l} = h^{ab} = \delta^{ab}$.

Definimos ahora los \emph{sĆ­mbolos de Christoffel} como
\begin{equation*}
\Gamma^{l}_{jk} = \frac{g^{li}}{2} \left( \frac{\partial g_{ij}}{\partial x^k} + \frac{\partial g_{ik}}{\partial x^j} - \frac{\partial g_{jk}}{\partial x^i} \right)
\end{equation*}
y concluimos que
\begin{equation*}
\frac{\partial^2 y^a}{\partial x^j \partial x^k} = \Gamma^i_{jk}\frac{\partial y^a}{\partial x^i}.
\end{equation*}

Si volvemos a derivar con respecto a $x^l$, obtenemos
\begin{equation*}
\frac{\partial^3 y^a}{\partial x^l \partial x^j \partial x^k} = \frac{\partial \Gamma^i_{jk}}{\partial x^l} \frac{\partial y^a}{\partial x^i} + \Gamma^i_{jk} \frac{\partial^2 y^a}{\partial x^l \partial x^i} = \left( \frac{\partial \Gamma^i_{jk}}{\partial x^l} + \Gamma^m_{jk} \Gamma^i_{ml} \right) \frac{\partial y^a}{\partial x^i}.
\end{equation*}
Como las derivadas mixtas parciales conmutan, tenemos que, para cada $a$, el siguiente tĆ©rmino se tiene que anular
\begin{equation*}
R^i_{jkl} \frac{\partial y^a}{\partial x^i} =\frac{\partial^3 y^a}{\partial x^k \partial x^j \partial x^l} - \frac{\partial^3 y^a}{\partial x^l \partial x^j \partial x^k} =\left( \frac{\partial \Gamma^i_{jl}}{\partial x^k} - \frac{\partial \Gamma^i_{jk}}{\partial x^l} +\Gamma^{m}_{jl} \Gamma^i_{mk} - \Gamma^m_{jk}\Gamma^i_{ml}  \right) \frac{\partial y^a}{\partial x^i} .
\end{equation*}
El tĆ©rmino
\begin{equation*}
R^{i}_{jkl} = \frac{\partial \Gamma^i_{jl}}{\partial x^k} - \frac{\partial \Gamma^i_{jk}}{\partial x^l} +\Gamma^{m}_{jl} \Gamma^i_{mk} - \Gamma^m_{jk}\Gamma^i_{ml}
\end{equation*}
define un tensor de tipo $(1,3)$, $R\in \mathcal{T}^1_3(M)$, denominado el \emph{tensor de curvatura de Riemann}.

Hemos demostrado lo siguiente:

\begin{prop}
  Si una variedad Riemanniana es localmente isomorfa al espacio euclidiano $\RR^n$ con su producto escalar estĆ”ndar, entonces se tiene
  \begin{equation*}
R=0.
  \end{equation*}
\end{prop}

En lo que sigue, veremos numerosos ejemplos de variedades para las cuales $R\neq0$.

\section{Curvatura extrĆ­nseca: Curvas planas}

\subsection{} Sea $E$ un plano euclidiano. Una \emph{curva no singular} en $E$ es una subvariedad regular $C\subset E$ de dimensiĆ³n 1 equipada con una orientaciĆ³n de la direcciĆ³n normal a $C$. El producto escalar en $E$ induce una mĆ©trica riemanniana en $C$. La elecciĆ³n de la orientaciĆ³n normal prescribe una elecciĆ³n de \emph{vector normal unitario} $\bm{n}$ que se define como el vector unitario (con respecto al producto escalar en $E$) en direcciĆ³n normal a $C$ y en el sentido prescrito por la orientaciĆ³n. Existen dos elecciones posibles para el \emph{vector tangente unitario}
$\bm{t}$
que es un vector unitario perpendicular a $\bm{n}$.

\subsection{} Una \emph{parametrizaciĆ³n por el arco} de $C$ en un abierto $U\subset C$ es una isometrĆ­a $\gamma:\RR\rightarrow U$. Si $\gamma$ es una parametrizaciĆ³n por el arco, con coordenada $s$, entonces
\begin{equation*}
\bm{t} = \frac{\partial}{\partial s}
\end{equation*}
determina una elecciĆ³n del vector tangente unitario. NĆ³tese que, para $s\in \RR$, tenemos
\begin{equation*}
\bm{t}(s):= \bm{t}|_{x(s)} = \left. \frac{\partial}{\partial s} \right|_{x(s)} = \frac{d\gamma}{ds}.
\end{equation*}

\subsection{Curvatura de una curva}
Por la simetrĆ­a del producto escalar y por regla de Leibniz, tenemos que
\begin{equation*}
2\left \langle \frac{d\bm{t}}{ds}, \bm{t} \right \rangle = \frac{d}{ds} \langle \bm{t},\bm{t}\rangle = 0.
\end{equation*}
Por tanto, podemos escribir
\begin{equation*}
\frac{d\bm{t}}{ds}(s) = \kappa(x(s)) \bm{n}|_{x(s)},
\end{equation*}
donde $\kappa:C\rightarrow \RR$ es una funciĆ³n denominada la \emph{curvatura} de $C\subset E$.

\subsection{} Podemos dar un argumento equivalente utilizando formas diferenciales con coeficientes en el espacio vectorial $V$ subyacente a $E$. Estos espacios $\Omega^k(C;V)$ de formas diferenciales $V$-valoradas se definen como los espacios de secciones de los fibrados $\Lambda^k T^*M \otimes_{\RR} V$. La diferencial exterior y demĆ”s operaciones extienden a $\Omega^k(C;V)$ de forma natural. Podemos ver los vectores tangente y normal unitarios $\bm{t}$ y $\bm{n}$ como funciones $C\rightarrow V$. Es decir, $\bm{t},\bm{n}\in \Omega^0(C;V)$. Por la regla de Leibniz, tenemos que
\begin{equation*}
\langle d \bm{t},\bm{n}\rangle + \langle\bm{t},d\bm{n} \rangle = 0,
\end{equation*}
de modo que existe una $1$-forma $\alpha \in \Omega^1(C)$ tal que
\begin{equation*}
d\bm{t} = \alpha \bm{n} \text{ y } d\bm{n} = -\alpha \bm{t}.
\end{equation*}

Denotamos por $\theta^1,\theta^2 :C\rightarrow V^*$ la base dual a $\bm{t},\bm{n}$. Tenemos que, si denotamos por $i:C\hookrightarrow E$  la inclusiĆ³n, entonces $i^* \theta^1 \in \Omega^1(C)$ es una $1$-forma en $C$, y podemos escribir
\begin{equation*}
\alpha = \kappa i^* \theta^1,
\end{equation*}
para una cierta funciĆ³n $\kappa:C\rightarrow \RR$. De nuevo, esta funciĆ³n es la curvatura de $C\subset E$.

\begin{ej}
  Sea $C$ la circunferencia de radio $R$ en el plano eucliano estĆ”ndar $\RR^2$. Esta curva admite una parametrizaciĆ³n local $\gamma:\RR\rightarrow C$, $\phi\mapsto (R\cos \phi, R\sin \phi)$. Si tomamos la orientaciĆ³n dada por la direcciĆ³n normal apuntando hacia dentro de la circunferencia, tenemos
  \begin{equation*}
\bm{n} = -\cos \phi \frac{\partial}{\partial x} - \sin \phi \frac{\partial}{\partial y}.
  \end{equation*}
Tomamos como vector tangente el que apunta en sentido antihorario a lo largo del cĆ­rculo
\begin{equation*}
\bm{t} = - \sin \phi \frac{\partial}{\partial x} + \cos \phi \frac{\partial}{\partial y}.
\end{equation*}
La base dual es
\begin{equation*}
\theta^1 = -\sin \phi\ d x + \cos \phi\ dy
\end{equation*}
\begin{equation*}
\theta^2 = -\cos \phi\ dx - \sin \phi\ dy.
\end{equation*}
De modo que
\begin{equation*}
i^* \theta^1  = -\sin{\phi}\ d(R\cos{\phi}) + \cos{\phi}\ d(R\sin{\phi}) = R \sin^2{\phi}\ d\phi + R \cos^2{\phi}\ d\phi = Rd\phi.
\end{equation*}
Por otra parte,
\begin{equation*}
d\bm{t} = -\cos \phi\ d\phi \frac{\partial}{\partial x} - \sin \phi\ d\phi \frac{\partial}{\partial y} = (d\phi) \bm{n}.
\end{equation*}
Por tanto,
\begin{equation*}
\alpha = d\phi = \frac{1}{R} i^* \theta^1,
\end{equation*}
de modo que la curvatura de la circunferencia de radio $R$ es
\begin{equation*}
\kappa = \frac{1}{R}.
\end{equation*}
\end{ej}

\subsection{AplicaciĆ³n de Weingarten} Otra forma de obtener la curvatura de una curva es mediante la aplicaciĆ³n de Weingarten. En primer lugar, observamos que como el vector normal unitario es unitario (por construcciĆ³n), Ć©ste determina una aplicaciĆ³n \[\bm{n}:C\rightarrow S(V),\] donde $S(V)\subset V$ es la circunferencia unidad, es decir
\begin{equation*}
S(V) = \left\{ v\in V: \langle v,v\rangle = 1 \right\}.
\end{equation*}
Esta aplicaciĆ³n es diferenciable, por construcciĆ³n, y su diferencial en un punto $p\in C$ es la llamada \emph{aplicaciĆ³n de Weingarten}
\begin{align*}
S_p=- d_p\bm{n} : T_pC  \rightarrow T_{\bm{n}(p)} S(V).
\end{align*}
El espacio tangente $T_{\bm{n}(p)}S(V)$ estĆ” generado por el vector tangente unitario $\bm{t}_p$, de modo que podemos escribir
\begin{equation*}
S_p(\bm{t}_p) = \lambda \bm{t}_p,
\end{equation*}
para un cierto nĆŗmero $\lambda \in \RR$, pero resulta que este nĆŗmero es precisamente $\kappa(p)$. En efecto,
\begin{equation*}
\lambda \bm{t}_p =-d_p\bm{n}(\bm{t}_p) =- (d\bm{n})_p(\bm{t}_p) = \alpha_p(\bm{t}_p) \bm{t}_p = \kappa (i^*\theta^1)_p(\bm{t}_p) \bm{t}_p = \kappa \bm{t}_p.
\end{equation*}

\subsection{Prescribir la curvatura} Consideramos ahora el siguiente problema:

Dada una $1$-variedad $C$ y una funciĆ³n $\kappa:C\rightarrow \RR$, construir una inmersiĆ³n $i:C\rightarrow E$, junto con una orientaciĆ³n de la direcciĆ³n normal a $i(C)$, de modo que $\kappa$ es la curvatura de $i(C)$.

La forma natural de atacar este problema es considerar el fibrado de referencias ortonormales $\cB_{\mathrm{O}}(E)\rightarrow E$. Un punto de este fibrado es una terna $(p;e_1,e_2)$, donde $p\in E$ es un punto y $(e_1,e_2)$ es una base ortonormal de $V$. En particular, en este caso el fibrado de referencias escinde de forma natural como un producto $\cB_{\mathrm{O}}(E)=E\times \cB_{\mathrm{O}}(V)$.
Lo que queremos construir es una inmersiĆ³n $i:C\rightarrow E$ junto con una elevaciĆ³n $\tilde{i}:C\rightarrow \cB_{\mathrm{O}}(E)$ de modo que el siguiente diagrama conmute
\begin{center}
  \begin{tikzcd}
    & \cB_{\mathrm{O}}(E) \ar{d}{\pi} \\
    C \ar{ru}{\tilde{i}} \ar{r}{i} & E.
  \end{tikzcd}
\end{center}
La elevaciĆ³n $\tilde{i}$ puede escribirse como una terna $\tilde{i}=(i,e_1,e_2)$. Fijemos una parametrizaciĆ³n local $\RR \rightarrow C$, con coordenada $s$, y denotemos por $\gamma,\bm{t},\bm{n}:\RR\rightarrow E$ las composiciones de $i$, $e_1$ y $e_2$, respectivamente, con la parametrizaciĆ³n. La terna $(\gamma,\bm{t},\bm{n})$ ha de ser una soluciĆ³n de la ecuaciĆ³n diferencial
\begin{equation*}
d\begin{pmatrix}
\gamma & \bm{t} & \bm{n}
\end{pmatrix} = \begin{pmatrix}
\gamma & \bm{t} & \bm{n}
\end{pmatrix}
\begin{pmatrix}
  0 & 0 & 0 \\
  ds & 0 & -\kappa ds \\
  0 & \kappa ds & 0
\end{pmatrix}.
\end{equation*}

Denotamos
\begin{equation*}
  A(s)=
\begin{pmatrix}
  0 & 0 & 0 \\
  ds & 0 & -\kappa(s) \\
  0 & \kappa(s) & 0
\end{pmatrix}.
\end{equation*}
Esta matriz $A(s)$ ha de interpretarse como un campo vectorial dependiente del parĆ”metro $s$ en la variedad $\cB_{\mathrm{O}}(E)$. En efecto, si $P(s)$ es una curva en $\mathrm{O}_2(\RR)$ con $P(0)=\mathbb{I}$, entonces $P^T P =\mathbb{I}$. Tenemos entonces
\begin{equation*}
0=\left. \frac{d}{ds} \right|_{s=0} (P^T P) = [P'(s)^T P(s) + P(s)^T P'(s)]_{s=0} = P'(0)^T + P'(0).
\end{equation*}
De modo que $T_{\mathbb{I}} O_2(\RR)$ es la recta de las matrices $2\times 2$ antisimĆ©tricas sobre $\RR$. En particular, la matriz
\begin{equation*}
\begin{pmatrix}
  0 & - \kappa \\
  \kappa & 0
\end{pmatrix}
\end{equation*}
es una de estas matrices, de modo que $A(s)$ puede verse como un campo en $\cB_{\mathrm{O}}(E)$ dependiente de $s$.

El teorema de existencia y unicidad de soluciones para sistemas de ecuaciones diferenciales de orden $1$ implica que, dados un punto inicial $p\in C$ y unos vectores unitarios perpendiculares $\bm{t}_0,\bm{n}_0 \in E$, existen un $\varepsilon > 0$ y unas Ćŗnicas aplicaciones diferenciables $\gamma,\bm{t},\bm{n}:(-\varepsilon,\varepsilon)\rightarrow E$ con $\gamma(0)=p$, $\bm{t}(0)=\bm{t}_0$ y $\bm{n}(0)=\bm{n}_0$ y tales que
\begin{equation*}
\frac{d}{ds}\begin{pmatrix}
\gamma(s) & \bm{t}(s) & \bm{n}(s)
\end{pmatrix} = \begin{pmatrix}
\gamma(s) & \bm{t}(s) & \bm{n}(s)
\end{pmatrix} A(s).
\end{equation*}
AdemĆ”s, esta soluciĆ³n se puede prolongar a un intervalo maximal $I \subset \RR$.

\section{Superficies en el espacio}

\subsection{} Sea $E$ un espacio euclidiano $3$-dimensional. Una \emph{superficie no singular} en $E$ es una subvariedad regular $S\subset E$ de dimensiĆ³n $2$ equipada con una orientaciĆ³n de la direcciĆ³n normal a $S$. El producto escalar en $E$ induce una mĆ©trica riemanniana $g$ en $S$; clĆ”sicamente, a esta mĆ©trica riemanniana se le llama la \emph{primera forma fundamental}, y se denota por $g_p=I_p$. Igual que en curvas, la orientaciĆ³n normal determina un vector normal unitario $\bm{n}$. La correspondiente aplicaciĆ³n
\begin{equation*}
\bm{n}: S \rightarrow S(E),
\end{equation*}
donde $S(E)\subset E$ es la esfera unidad en $E$, se denomina la \emph{aplicaciĆ³n de Gauss}.
De nuevo, podemos considerar la \emph{aplicaciĆ³n de Weingarten}, para cada $p\in S$
\begin{equation*}
S_p = -d_p \bm{n}: T_pS \rightarrow T_p S(E).
\end{equation*}
De nuevo, la elecciĆ³n de $\bm{n}$ induce un isomorfismo
\begin{equation*}
T_pS \cong \ker \langle\bm{n},-\rangle \cong T_p S(E),
\end{equation*}
de modo que podemos ver la aplicaciĆ³n de Weingarten como un operador
\begin{equation*}
S_p:T_pS\rightarrow T_pS.
\end{equation*}

\subsection{Curvatura, operador de forma y segunda forma fundamental} Una forma de introducir una nociĆ³n de curvatura para $S$ es considerar, para cada $p\in S$, la aplicaciĆ³n
\begin{equation*}
\kappa_p: \mathbb{P}(T_p S) \rightarrow \RR,
\end{equation*}
que asigna a cada recta $L\subset T_pS$ la curvatura de la curva $S\cap \Pi(L)$, obtenida como la intersecciĆ³n de $S$ con el plano $\Pi(L)\subset E$ que contiene a $L$ y pasa por $p$.

Por el teorema de la funciĆ³n inversa, podemos ver localmente la superficie $S$ como la grĆ”fica de una funciĆ³n $F:T_pS\rightarrow \RR$ con $F(0)=0$ y $d_0 F=0$. Se denomina la \emph{segunda forma fundamental} al hessiano
\begin{equation*}
II_p:=\mathrm{Hess}_p F: T_pS \times T_p S \rightarrow \RR.
\end{equation*}
La funciĆ³n $\kappa_p$ mide la segunda derivada de $F$, en el sentido de que se tiene, para cada recta $L\subset T_pS$, y para cada vector unitario $v\in L$,
\begin{equation*}
II_p(v,v) = \kappa_p(L).
\end{equation*}

Como la primera forma fundamental $I_p=g_p$ es no degenerada, ha de existir un operador $S_p:T_pS \rightarrow T_pS$, llamado el \emph{operador de forma}, de modo que
\begin{equation*}
II_p(v,w) = I_p(v,S_p(w)).
\end{equation*}

\begin{lemma}
El operador de forma $S_p$ coincide con la aplicaciĆ³n de Weingarten.
\end{lemma}

\begin{proof}
  Tomamos ``coordenadas euclidianas'' $x$ y $y$ en el espacio afĆ­n tangente $p+T_p S \subset E$ y definimos $F=F(x,y)$ de modo que en un entorno de $p$, tenemos que $S$ es la grĆ”fica de $F$. Entonces
  \begin{equation*}
\bm{n}(x,y) = \frac{(\frac{\partial F}{\partial x}, \frac{\partial F}{\partial y}, 1)}{\lVert (\frac{\partial F}{\partial x}, \frac{\partial F}{\partial y}, 1)\rVert},
  \end{equation*}
  cuya matriz jacobiana es
  \begin{equation*}
    J_{(x,y)} \bm{n}=
-\left. \begin{pmatrix}
\partial_{xx} F & \partial_{xy} F \\
\partial_{yx} F & \partial_{yy} F \\
\end{pmatrix} \right|_p = - \mathrm{Hess}_p F,
  \end{equation*}
  que es la matriz asociada a $II_p$ en estas coordenadas euclidianas.
\end{proof}

\subsection{} Como la segunda forma fundamental es simĆ©trica, el operador de forma $S_p$ es autoadjunto, por lo que es diagonalizable y tiene autovalores reales $\lambda_1,\lambda_2 \in \RR$. Distinguimos dos casos:
\begin{enumerate}
\item Si $\lambda_1=\lambda_2=\lambda$, entonces $S_p = \lambda \mathrm{id}_{T_p S}$. Decimos entonces que $p$ es un \emph{punto umbilical}.
  \item Si $\lambda_1\neq \lambda_2$ (supongamos, sin pĆ©rdida de generalidad que $\lambda_1>\lambda_2$), entonces el espacio $T_pS$ descompone ortogonalmente como $T_pS=L_1\oplus L_2$, donde los $L_i$ son los autoespacios, llamados \emph{lĆ­neas principales}, con autovalores $\lambda_i$, llamados \emph{curvaturas principales}. Se tiene que $\lambda_1$ es el mĆ”ximo de la funciĆ³n $\kappa_p$, que se alcanza en $L_1$, mientras que $\lambda_2$ es su mĆ­nimo, que se alcanza en $L_2$.
\end{enumerate}

Al cambiar la orientaciĆ³n normal, se intercambian $\kappa_p \mapsto -\kappa_p$ y $\lambda_i\mapsto -\lambda_i$. Se denomina \emph{curvatura media} a la funciĆ³n
\begin{equation*}
H(p) := \frac{\lambda_1 + \lambda_2}{2} = \frac{1}{2} \mathrm{tr}(S_p).
\end{equation*}
Al cambiar la orientaciĆ³n normal, tenemos $H\mapsto -H$. Por otra parte, tenemos la \emph{curvatura de Gauss}
\begin{equation*}
K(p) := \lambda_1 \lambda_2 = \det(S_p),
\end{equation*}
que permanece invariante al cambiar la orientaciĆ³n normal.

\begin{thm}[Gauss--Bonnet]
  Sea $S\subset E$ una superficie no singular con mĆ©trica riemanniana inducida $g$ y curvatura de Gauss $K:S\rightarrow \RR$. Se tiene que
  \begin{equation*}
\int_S K d\mathrm{vol}_g = 2\pi \chi(S),
  \end{equation*}
  donde $\chi(S)$ es la caracterĆ­stica de Euler de $S$.
\end{thm}

\subsection{} Recordemos que la caracterĆ­stica de Euler de una superficie $S$ se definĆ­a como \[\chi(S)=C-A+V,\] donde $C$, $A$ y $V$ son el nĆŗmero de caras, aristas y vĆ©rtices, respectivamente, de cualquier triangulaciĆ³n o, mĆ”s generalmente, de cualquier descomposiciĆ³n en celdas de $S$. TambiĆ©n coincide con $2-2g(S)$, donde $g(S)=\frac{1}{2}\dim H^1(S,\RR)$ es el gĆ©nero de $S$.

\begin{ej}
  Supongamos que $S\subset E$ es una esfera de radio $R$. Todos los puntos de la esfera son umbilicales, con curvatura de Gauss igual a $1/R^2$. Por otra parte, el Ć”rea de la esfera de radio $R$ es $4\pi R^2$. Por tanto,
  \begin{equation*}
\int_S K d\mathrm{vol}_g = 4\pi R^2 \frac{1}{R^2} = 4\pi = 2\cdot 2\pi.
  \end{equation*}
En efecto, la caracterĆ­stica de Euler de la esfera es $2$.
\end{ej}

\begin{proof}[DemostraciĆ³n del teorema de Gauss--Bonnet]
  Recordemos que, si $M$ y $N$ son variedades compactas y orientadas de dimensiĆ³n $n$, podĆ­amos definir el grado de una aplicaciĆ³n diferenciable $\nu:N\rightarrow M$ por la fĆ³rmula
  \begin{equation*}
\int_N \nu^* \omega = \deg \nu \int_M \omega,
  \end{equation*}
  para cualquier $\omega \in \Omega^n(M)$. En nuestro caso, podemos tomar $\nu=\bm{n}:S\rightarrow S(V)$, y tenemos que
  \begin{equation*}
\bm{n}^* d\mathrm{vol}_{S(V)} = \det(d\bm{n}) d\mathrm{vol}_g = \det(-S) d\mathrm{vol}_g = -Kd\mathrm{vol}_g.
  \end{equation*}
  Por tanto,
  \begin{equation*}
\int_S K d\mathrm{vol}_g =- \int_S \bm{n}^* d\mathrm{vol}_{S(V)} = -\deg \bm{n} \int_S d\mathrm{vol}_{S(V)}=-4\pi \deg \bm{n}.
  \end{equation*}
Basta ver entonces que $\deg \bm{n} = -\chi(S)/2$.

Recordemos ahora que el grado $\deg \bm{n}$ puede computarse como
\begin{equation*}
\deg \bm{n} = \sum_{p \in \bm{n}^{-1}(q)} \mathrm{sgn} \det(d_p \bm{n}),
\end{equation*}
donde $q\in S(V)$ es cualquier valor regular. Ahora, recordemos que
\begin{equation*}
d_p \bm{n} = - \mathrm{Hess}_p F,
\end{equation*}
donde $F$ es cualquier funciĆ³n $F:T_pS\rightarrow \RR$ con $F(0)=0$ y $d_0F=0$ tal que $S$ es localmente su grĆ”fica.

El siguiente paso es considerar una funciĆ³n de altura $H:E\rightarrow \RR$. Para ello, consideramos un punto base $p_0\in E$, de modo que podemos identificar $E$ y $V$, y fijamos un vector unitario $u\in S(V)$. Definimos
\begin{equation*}
H(v) = \langle u,v\rangle.
\end{equation*}
Para cada nĆŗmero real $a\in \RR$, podemos considerar el conjunto de nivel
\begin{equation*}
S_a = H^{-1}(a) \cap S = \left\{ p \in S: H(p)=a \right\},
\end{equation*}
que es posiblemente vacĆ­o. De hecho, como $S$ es compacta, existe un $M>0$ tal que para $S_a=\varnothing$ para $a<-M$ y para $a>M$.
Los puntos crĆ­ticos de $H$ son aquellos $p\in S$ para los que
\begin{equation*}
\grad H (p) = 0.
\end{equation*}
Dado un punto crĆ­tico $p$ podemos tener uno de los siguientes casos
\begin{itemize}
\item $p$ es un mĆ­nimo de $H$, de modo que $\mathrm{sgn} \det(\mathrm{Hess}_p H) = 1$.
  \item $p$ es un mĆ”ximo de $H$, de modo que $\mathrm{sgn} \det(\mathrm{Hess}_p H) = 1$.
    \item $p$ es un punto de silla de $H$, de modo que $\mathrm{sgn} \det(\mathrm{Hess}_p H) = -1$.
\end{itemize}
Sea $\varphi:\RR \times S \rightarrow S$ el flujo generado por el campo $-\grad H$.
Utilizando $\varphi$, podemos construir una descomposiciĆ³n en celdas de $S$ de la siguiente forma:
\begin{itemize}
  \item para cada mĆ­nimo $p$ de $H$, tenemos una $2$-celda
        \begin{equation*}
\Delta_{p} = \left\{ q \in S: \lim_{t\rightarrow \infty} \varphi(t,q) =p  \right\},
        \end{equation*}
  \item los vĆ©rtices (las $0$-celdas) son los mĆ”ximos de $H$,
\item a cada punto de silla $p$ le corresponde una $1$-celda
        \begin{equation*}
A_p = \left\{ q \in S: \lim_{t\rightarrow \infty} \varphi(t,q) = p \right\}.
        \end{equation*}
\end{itemize}
Por tanto, concluimos que
\begin{equation*}
\chi(S) = \sum_{p\in \mathrm{Crit}(H)} \mathrm{sgn} \det(\mathrm{Hess}_p H).
\end{equation*}

A priori, habrĆ­a que tener un poco mĆ”s de cuidado en la elecciĆ³n de $H$, ya que podrĆ­a pasar que tuviera infinitos puntos crĆ­ticos, o que las $1$-celdas asĆ­ definidas unan puntos de silla con puntos de silla, y no puntos de silla con mĆ”ximos, como deseamos. Con argumentos tĆ­picos de topologĆ­a diferencial se puede modificar ligeramente la funciĆ³n $H$ para evitar que se de esa situaciĆ³n patolĆ³gica. En tĆ©rminos tĆ©cnicos, lo que se busca es que $\grad H$ sea un \textit{campo de Morse--Smale}.

Por otra parte, tenemos que $\nabla H = u$, de modo que los puntos crĆ­ticos $p$ de $H$ son aquellos para los cuales
\begin{equation*}
\langle u, v \rangle = 0
\end{equation*}
para todo $v\in T_pS$ o, lo que es lo mismo
\begin{equation*}
\mathrm{Crit}(H) = \bm{n}^{-1}(u) \cup \bm{n}^{-1}(-u).
\end{equation*}
AdemĆ”s, si tomamos coordenadas locales en torno a un punto crĆ­tico $p$ con $p=0$, $v=(0,0,1)$ y $S$ dada por $z=F(x,y)$, entonces
\begin{equation*}
F = H|_S,
\end{equation*}
de modo que $\mathrm{Hess}_p(H) = \mathrm{Hess}_p(F)$. Concluimos que
\begin{align*}
\chi(S) &= \sum_{p\in \mathrm{Crit}(H)} \mathrm{sgn} \det(\mathrm{Hess}_p H) \\ &= \sum_{p \in \bm{n}^{-1}(u)} \mathrm{sgn} \det(\mathrm{Hess}_p F) + \sum_{p \in \bm{n}^{-1}(-u)} \mathrm{sgn} \det(\mathrm{Hess}_p F) = -2\deg(\bm{n}),
\end{align*}
como querĆ­amos probar
\end{proof}

\section{GeometrĆ­a riemanniana intrĆ­nseca}
\subsection{} Vamos a concluir nuestro estudio de las superficies viendo quĆ© aspectos de la geometrĆ­a de superficies en el espacio $S\subset E$ son intrĆ­nsecos a la superficie $S$, con su estructura riemanniana (dada por la primera forma fundamental) y no dependen del espacio ambiente. En primer lugar, veremos que el tensor de curvatura de Riemann en superficies se reduce simplemente a la curvatura de Gauss. Por tanto, la curvatura de Gauss es intrĆ­nseca. Este es el contenido del \emph{teorema egregio de Gauss}. Seguidamente, estudiaremos la \emph{derivada covariante}, que nos proporcionarĆ” una interpretaciĆ³n geomĆ©trica de los sĆ­mbolos de Christoffel.

\begin{thm}[Teorema egregio de Gauss]
  Dada una superficie $S\subset E$ y un punto $p\in S$, se tiene
  \begin{equation*}
K(p) = R^1_{212}(p),
  \end{equation*}
donde $K$ es la curvatura de Gauss y $R^1_{212}$ es el tensor de curvatura de Riemann. Por tanto $K$ es intrĆ­nseca a la superficie.
\end{thm}

\begin{proof}
  En un entorno de $p$ podemos ver la superficie $S$ como la grĆ”fica de una funciĆ³n $(x^1,x^2)\mapsto (x^1,x^2,F(x^1,x^2))$, de modo que tenemos
  \begin{equation*}
\partial_1|_{(x^1,x^2)} = (1,0,\partial_1 F) \text{ y } \partial_2|_{(x^1,x^2)} = (1,0,\partial_2 F).
  \end{equation*}
  La mĆ©trica riemanniana se escribe entonces como
  \begin{equation*}
g_{11} = 1 + (\partial_1F)^2, \ g_{12}= \partial_1 F \partial_2 F, \ g_{22}=1 + (\partial_2F)^2.
  \end{equation*}
  Si denotamos $\Delta = 1 + (\partial_1F)^2 + (\partial_2F)^2$, tenemos
  \begin{equation*}
g^{11} = (1 + (\partial_2F)^2)/\Delta, \ g^{12}= -\partial_1 F \partial_2 F/\Delta, \ g^{22}=(1 + (\partial_1F)^2)/\Delta.
  \end{equation*}
 Calculamos tambiĆ©n
  \begin{equation*}
A_{ijk} := \partial_k g_{lj} + \partial_j g_{lk} -\partial_l g_{jk} = 2 \partial_i F \partial_{jk} F.
  \end{equation*}
  Recordemos ahora que, por construcciĆ³n, $F$ y sus derivadas parciales se anulan en $(0,0)$, de modo que los $A_{ijk}(0)=0$ y, por tanto $\Gamma^i_{jk}(0) = 0$. En efecto, recordemos que los sĆ­mbolos de Christoffel vienen dados por la fĆ³rmula
  \begin{equation*}
\Gamma^i_{jk} = \frac{1}{2} g^{il} A_{ljk}.
  \end{equation*}
  Por tanto,
  \begin{equation*}
R^1_{212}(0,0) = \partial_1 \Gamma^1_{22}(0,0) - \partial_2\Gamma^1_{21}(0,0).
  \end{equation*}
  Ahora,
  \begin{equation*}
\Gamma^1_{22} = \frac{1}{2} g^{11} A_{122} + \frac{1}{2} g^{12} A_{222} = \frac{1}{\Delta}((1+\partial_2 F) \partial_1F \partial_{22} F - \partial_1 F (\partial_2 F)^2 \partial_{22}F) = \frac{1}{\Delta} \partial_1F \partial_{22}F.
  \end{equation*}
  Similarmente, vemos que
  \begin{equation*}
\Gamma^1_{21} = \frac{1}{\Delta} \partial_1 F \partial_{21} F.
  \end{equation*}
  Por tanto,
  \begin{equation*}
R^1_{212}(0,0) = (\partial_{11}F \partial_{22}F - \partial_{12}F\partial_{21}F)|_{(0,0)} = \det \mathrm{Hess}_{(0,0)} F = K(p),
  \end{equation*}
como querĆ­amos probar.
\end{proof}

\subsection{} Procedemos ahora a dar una interpretaciĆ³n geomĆ©trica de los sĆ­mbolos de Christoffel. Comenzamos estudiando la nociĆ³n de \emph{paralelismo}. En un espacio euclidiano $E$, dado un campo vectorial $X:E\rightarrow V$, podemos calcular sus derivadas direccionales a lo largo de una direcciĆ³n $v_p \in T_pE$, definiendo
\begin{equation*}
D_{v_p} X : = \lim_{t\rightarrow 0} \frac{X(p+tv_p)-X(p)}{t}.
\end{equation*}
Si $\gamma:(-\varepsilon,\varepsilon)\rightarrow E$ es una curva con $\gamma(0)=p$ y $\gamma'(0)=v_p$, entonces
\begin{equation*}
D_{v_p} X = \left. \frac{d}{dt}\right|_{t=0} X(\gamma(t)).
\end{equation*}
Sin embargo, cuando consideramos una superficie $S\subset E$, tomar la derivada direccional de un campo vectorial en general nos puede dar algo que ``se salga'' de la superficie. Una forma de remediarlo es proyectar esta derivada direccional a $TS$.

\begin{defn}
La \emph{derivada covariante} $\nabla_{v_p}X$ de un campo vectorial $X\in \fX(S)$ a lo largo de una direcciĆ³n $v_p\in T_pS$ es la proyecciĆ³n de $D_{v_p}X \in V$ a $T_pS$.
\end{defn}

\begin{defn}
Sea $\gamma:(a,b)\rightarrow S$ una curva en la superficie $S$ y $X$ un campo vectorial a lo largo de $\gamma$. Es decir, $X$ es una secciĆ³n del fibrado imagen recĆ­proca $\gamma^* TS \rightarrow (a,b)$, con fibras $(\gamma^* TS)_t = T_{\gamma(t)} S$ para $t\in (a,b)$. Se dice que $X$ es un campo \emph{paralelo} a lo largo de $\gamma$ si $\nabla_{\gamma'} X = 0$. Si $\nabla_{\gamma'} \gamma' = 0$, es decir, si $\gamma'$ es paralelo a lo largo de $\gamma$, se dice que $\gamma$ es una \emph{geodĆ©sica}.
\end{defn}

\subsection{} Las geodĆ©sicas son las curvas que no tienen aceleraciĆ³n a lo largo de ellas mismas, de modo que la Ćŗnica aceleraciĆ³n que aparece es la aceleraciĆ³n normal a la superficie, que es la aceleraciĆ³n necesaria para mantenerse en la superficie. Por ejemplo, las geodĆ©sicas de la esfera son las circunferencias maximales.

Las siguientes propiedades se siguen fĆ”cilmente de las definiciones.

\begin{lemma}
\begin{enumerate}
\item La aplicaciĆ³n $\nabla X: T_pS \rightarrow T_pS$, $v_p \mapsto \nabla_{v_p}X$ es lineal.
\item La aplicaciĆ³n $\nabla_{v_p}: \fX(S) \rightarrow T_pS$, $X \mapsto \nabla_{v_p}X$ satisface la regla de Leibniz
        \begin{equation*}
\nabla_{v_p}(fX) = (v_p\cdot f) X + f \nabla_{v_p}X.
        \end{equation*}
\item $\nabla_{v_p}(X+X')=\nabla_{v_p}(X) + \nabla_{v_p}(X')$.
\item $v_p \cdot \langle X, X' \rangle = \langle \nabla_{v_p}X,X'\rangle + \langle X, \nabla_{v_p}X'\rangle$.
\end{enumerate}
\end{lemma}

\begin{thm} [Teorema fundamental de la geometrĆ­a riemanniana. Caso de superficies]
  La derivada covariante es intrĆ­nseca a la superficie $S$ y estĆ” determinada Ćŗnicamente por la mĆ©trica riemanniana $g$. Por tanto, la derivada covariante determina un operador
  \begin{align*}
    \nabla: \Omega^0(S;TS) & \longrightarrow \Omega^1(S;TS) \\
      X & \longmapsto (v\mapsto \nabla_v X),
  \end{align*}
  denominado la \emph{conexiĆ³n de Levi-Civita}.
\end{thm}

\begin{proof}
  Tomamos coordenadas $(x^1,x^2,F(x^1,x^2))$ igual que antes, y escribimos $X=X^i\partial_i$, de modo que $X=(X^1,X^2,X^i\partial_i F)$. Por la regla de Leibniz
  \begin{equation*}
D X = (dX^1, dX^2, \partial_i F dX^i + X^i d(\partial_iF)).
  \end{equation*}
  Por tanto, $\nabla X = (dX^1,dX^2,0)$ o, lo que es lo mismo
  \begin{equation*}
\nabla X = dX^i \partial_i,
  \end{equation*}
  de modo que $\nabla \partial_i = 0$ en $p$.

  Tomemos ahora unas coordenadas arbitrarias $y^1, y^2$. Podemos escribir $\partial_{y^a}=\frac{\partial x^i}{\partial y^a} \partial_{x^i}$, de modo que, en $p$,
  \begin{equation*}
\nabla \partial_{y^a} = \frac{\partial^2 x^i}{\partial y^b \partial y^a} dy^b \partial_{x^i} = \frac{\partial^2 x^i}{\partial y^b \partial y^a} \frac{\partial y^c}{\partial x_i}dy^b \partial_{y^c}.
  \end{equation*}
  Denotamos
  \begin{equation*}
Q^c_{ab} = \frac{\partial^2 x^i}{\partial y^b \partial y^a} \frac{\partial y^c}{\partial x_i},
  \end{equation*}
  de modo que
  \begin{equation*}
\nabla \partial_{y^a} = Q^c_{ab} dy^b \partial_{y^c}.
  \end{equation*}
Basta entonces probar que $Q^c_{ab}=\Gamma^c_{ab}$. Como los sĆ­mbolos de Christoffel son intrĆ­nsecos a la mĆ©trica, tambiĆ©n lo serĆ” $Q^c_{ab}$, lo que probarĆ” que $\nabla$ es intrĆ­nseca.

En $p$, tenemos que
\begin{equation*}
g_{ab} = \langle \partial_{y^a},\partial_{y^b} \rangle = \frac{\partial x^i}{\partial y^a}\frac{\partial x^j}{\partial y^b} g_{ij} = \sum_i \frac{\partial x^i}{\partial y^a}\frac{\partial x^i}{\partial y^b},
\end{equation*}
de modo que
\begin{equation*}
\frac{\partial g_{ab}}{\partial y^c} = \sum_i \frac{\partial^2 x^i}{\partial y^c \partial y^a}\frac{\partial x^i}{\partial y^b} + \frac{\partial x^i}{\partial y^a}\frac{\partial^2 x^i}{\partial y^c \partial y^b}.
\end{equation*}
Tenemos entonces que
\begin{equation*}
A_{dab} = 2 \sum_i \frac{\partial^2 x^i}{\partial y^a \partial y^b} \frac{\partial x^i}{\partial y^d},
\end{equation*}
mientras que
\begin{equation*}
g^{cd} = \sum_j \frac{\partial y^c}{\partial x^j} \frac{\partial y^d}{\partial x^j}.
\end{equation*}
Por tanto,
\begin{align*}
\Gamma^c_{ab} &= \frac{1}{2} g^{cd} A_{dab} = \sum_{ij} \frac{\partial y^c}{\partial x^j} \frac{\partial y^d}{\partial x^j}\frac{\partial x^i}{\partial y^d}\frac{\partial^2 x^i}{\partial y^a \partial y^b} \\ &= \sum_{ij} \delta^i_j \frac{\partial y^c}{\partial x^j}\frac{\partial^2 x^i}{\partial y^a \partial y^b} = \frac{\partial y^c}{\partial x^i}\frac{\partial^2 x^i}{\partial y^a \partial y^b}=Q^c_{ab},
\end{align*}
como querĆ­amos probar.
\end{proof}

\begin{defn}
Sea $M$ una variedad diferenciable y $E\rightarrow M$ un fibrado vectorial (real). Una \emph{conexiĆ³n} en $E$ es una aplicaciĆ³n $\RR$-lineal
\begin{equation*}
\nabla: \Omega^0(M;E) \rightarrow \Omega^1(M;E)
\end{equation*}
que satisface la regla de Leibniz
\begin{equation*}
\nabla(fs) = df \otimes s + f \nabla s,
\end{equation*}
para cualesquiera $f \in C^\infty(M)$ y $s\in \Omega^0(M;E)$. El operador $\nabla$ extiende de forma natural a todas las potencias tensoriales de $E$, y determina conexiones en todas ellas, que tambiĆ©n denotamos por $\nabla$.

Cuando $E=TM$ es el fibrado tangente de $M$, una conexiĆ³n en $E$ se denomina una \emph{conexiĆ³n afĆ­n}. En ese caso, podemos escribir $\nabla$ equivalentemente como
\begin{equation*}
\nabla: \fX(M) \times \fX(M) \rightarrow \fX(M),\ (X,Y) \mapsto \nabla_XY.
\end{equation*}
  Se define la \emph{torsiĆ³n} de una conexiĆ³n afĆ­n $\nabla$ en $M$ como el operador $T:\fX(M)\times \fX(M)\rightarrow \fX(M)$ definido como
  \begin{equation*}
T(X,Y) = \nabla_X Y - \nabla_Y X - [X,Y].
  \end{equation*}
  Si $T=0$, entonces se dice que la conexiĆ³n $\nabla$ es \emph{libre de torsiĆ³n}.
\end{defn}

\begin{thm}[Teorema fundamental de la geometrĆ­a riemanniana]
Dada una variedad riemanniana $(M,g)$, existe una Ćŗnica conexiĆ³n afĆ­n y libre de torsiĆ³n $\nabla$ en $M$ tal que
\begin{equation*}
\nabla g = 0.
\end{equation*}
Es decir, tal que $\nabla$ preserva la mĆ©trica $g$. Dicha conexiĆ³n se denomina la \emph{conexiĆ³n de Levi-Civita}.
\end{thm}

\section{La conexiĆ³n de Levi-Civita}
\subsection{ExpresiĆ³n de una conexiĆ³n en coordenadas} Sea $M$ una variedad diferenciable y $E\rightarrow M$ un fibrado vectorial. Denotamos por $\cB(E)\rightarrow M$ el correspondiente fibrado de referencias, cuya fibra en un punto $x\in M$ es el conjunto $\cB(E)_x=\cB(E_x)$ de las bases de $E_x$. Cada fibra $\cB(E)_x$ es un torsor bajo el grupo $\GL_r(\RR)$, donde $r$ es el rango de $E$. Una \emph{referencia local} de $E$ en un abierto $U\subset M$ es una secciĆ³n local $\bm{e}: U \rightarrow \cB(E)$ o, lo que es lo mismo, una tupla $\bm{e}=(e_1,\dots,e_r)$ de secciones $e_i:U\rightarrow E$ que determina una base de $E_x$ en cada punto $x\in U$. Cualquier otra secciĆ³n $s$ de $E$ en $U$ se escribe como
\begin{equation*}
s = s^i e_i.
\end{equation*}

Supongamos ahora que $\nabla: \Omega^0(M;E) \rightarrow \Omega^1(M;E)$ es una conexiĆ³n en $E$. En la referencia $\bm{e}$ la conexiĆ³n estĆ” determinada por unas $1$-formas $A^i_{j} \in \Omega^1(M)$ definidas como
\begin{equation*}
\nabla(e_i) =A^j_i  e_j.
\end{equation*}
Agrupando las $A^i_j$, obtenemos una $1$-forma $A\in \Omega^1(U;\End E)$ que cumple
\begin{equation*}
\nabla(\bm{e}) = A \bm{e}.
\end{equation*}
A su vez, tenemos
\begin{equation*}
\nabla(s^i e_i) = d(s^i) e_i + s^jA^i_je_i =(\partial_k s^i dx^k+ s^jA^i_{j}) e_i .
\end{equation*}
La $1$-forma $A$ descompone como
\begin{equation*}
A = A_i dx^i,
\end{equation*}
y en particular cada $1$-forma $A^i_j$ descompone como
\begin{equation*}
A^i_j=A^i_{kj} dx^k,
\end{equation*}
de modo que
\begin{equation*}
\nabla(s^i) =(\partial_k s^i + s^jA^i_{kj}) dx^k .
\end{equation*}

Si se toma una referencia distinta $\bm{e}'$, relacionada con $\bm{e}$ por una aplicaciĆ³n diferenciable $h:U\rightarrow \GL_r(\RR)$, de modo que $\bm{e}'=\bm{e} h$, entonces podemos escribir
\begin{equation*}

\end{equation*}

\end{equation*}

\end{equation*}

\end{equation*}


\subsection{ConexiĆ³n de Levi-Civita} Pasamos a construir la conexiĆ³n de Levi-Civita en una variedad riemanniana $(M,g)$. Para ello, tomamos unas coordenadas locales $\bm{x}:U\rightarrow \RR^n$ y consideramos los correspondientes sĆ­mbolos de Christoffel

\end{equation*}

\end{equation*}
Por tanto, los sĆ­mbolos de Christoffel determinan los coeficientes de una conexiĆ³n afĆ­n en $M$. AdemĆ”s, esta conexiĆ³n es libre de torsiĆ³n ya que $\Gamma_{ij}^k=\Gamma_{ji}^k$. La conexiĆ³n asĆ­ definida se denomina la \emph{conexiĆ³n de Levi-Civita}. Podemos probar entonces el teorema fundamental.

\begin{thm}[Teorema fundamental de la geometrĆ­a riemanniana]
La Ćŗnica conexiĆ³n afĆ­n $\nabla$ en $M$ libre de torsiĆ³n y con $\nabla g  = 0$ es la conexiĆ³n de Levi-Civita.
\end{thm}

\begin{proof}
  En efecto, si $\nabla$ es una conexiĆ³n con $\nabla g=0$, tenemos

  \end{equation*}
  Por tanto,
  \begin{equation*}

  \end{equation*}
  Ahora, si $\nabla$ es libre de torsiĆ³n, tenemos que $A_{kij}=A_{ikj}$, de modo que

  \end{equation*}
  luego
  \begin{equation*}

  \end{equation*}
Esto prueba que $\nabla$ es la conexiĆ³n de Levi-Civita.
\end{proof}

\begin{defn}
  Un \emph{Ć”lgebra de Lie} (real) es un espacio vectorial real $\fg$ equipado de un \emph{corchete de Lie}: una forma bilineal antisimĆ©trica
  \begin{equation*}

  \end{equation*}
(es decir, con $[x,y]=-[y,x]$) y que satisface la \emph{identidad de Jacobi}
\begin{equation*}
[[x,y],z] + [[y,z],x] + [[z,x], y] = 0.
\end{equation*}
\end{defn}

\begin{prop}

\end{prop}

\subsection{} Supongamos que $\fg$ es un Ć”lgebra de Lie de dimensiĆ³n finita y que $x_1,\dots,x_n$ es una base de $\fg$. En ese caso, deben existir unas constantes $C^i_{jk}\in \RR$ tales que
\begin{equation*}

\end{equation*}
Dichas constantes se denominan las \emph{constantes de estructura} de $\fg$ con respecto a la base $x_1,\dots,x_n$. De la antisimetrĆ­a del corchete, se deduce que $C^i_{jk}=-C^i_{kj}$, mientras que de la identidad de Jacobi se deduce que
\begin{equation*}

\end{equation*}

\begin{ej}
  El ejemplo mĆ”s simple de Ć”lgebra de Lie es el Ć”lgebra $\mathfrak{gl}_n(\RR)$ formado por las matrices $n\times n$ reales, donde el corchete de Lie estĆ” dado por el conmutador

  \end{equation*}
  Una base de $\mathfrak{gl}_n(\RR)$ estĆ” dada por las matrices elementales $E_{ij}$, que tienen un $1$ en la entrada $(i,j)$ y un $0$ en el resto de entradas de modo que $\mathfrak{gl}_n(\RR)$ tiene dimensiĆ³n $n^2$. Si multiplicamos dos de estas matrices, tenemos

  \end{equation*}
  de modo que
  \begin{equation*}

  \end{equation*}
  Si escribimos
  \begin{equation*}

  \end{equation*}
  tenemos que las constantes de estructura son
  \begin{equation*}

  \end{equation*}
  ExplĆ­citamente

      -E_{kj} & \text{ si } l=i, j\neq k, \\
      E_{ii}-E_{jj} & \text{ si } j=k, l=i, \\

\end{cases}
  \end{equation*}
\end{ej}


\begin{ej}

  \end{equation*}
donde $\epsilon_{ij}^k$ es el sĆ­mbolo de Levi-Civita, que asocia a cada permutaciĆ³n $(i,j,k)$ de $(1,2,3)$, el signo de la permutaciĆ³n. Tenemos entonces que las constantes de estructura de $\mathfrak{so}_3(\RR)$ con respecto a esta base son $C_{ij}^k = \epsilon_{ij}^k$.
\end{ej}

\subsection{} Dada un Ć”lgebra de Lie $\fg$, podemos definir una aplicaciĆ³n $d:\fg^* \rightarrow \Lambda^2\fg^*$, tomando

\section{Grupos de Lie}
\begin{defn}
Un \emph{grupo de Lie} es una variedad diferenciable $G$ equipada con sendas aplicaciones diferenciables
\begin{equation*}

\end{equation*}
denominada la \emph{multiplicaciĆ³n}, e
\begin{equation*}

\end{equation*}
denominada la \emph{inversiĆ³n}, de modo que $(G,m)$ determina una estructura de grupo, con inversa dada por $i$.

[Otra forma de decir lo mismo es decir que $G$ es un grupo en la categorĆ­a de las variedades diferenciables.]
\end{defn}

\begin{ej}

\begin{itemize}
\item Si $V$ es un espacio vectorial de dimensiĆ³n finita, entonces $(V,+)$ es un grupo de Lie (conmutativo).
  \item El \emph{grupo general lineal}  $\GL(V)=\mathrm{Aut}(V)$ de automorfismos de $V$ es un grupo de Lie, no conmutativo si $\dim V > 1$. Cuando $V=\RR^n$, tenemos que $\GL_n(\RR):=\GL(\RR^n)$ es el grupo de las matrices invertibles $n\times n$. Como variedad, es simplemente el subconjunto abierto del espacio de las matrices $n\times n$, que es isomorfo a $\RR^{n^2}$, determinado por la condiciĆ³n $\det \neq 0$.         \end{equation*}

        entonces $\mathrm{Sp}_n(\RR):=\mathrm{Sp}(\RR^{2n})$ es el grupo de las matrices $2n \times 2n$ que cumplen la condiciĆ³n
        \begin{equation*}

        \end{equation*}
        donde $J$ es la matriz $2n \times 2n$ definida como
        \begin{equation*}

          \begin{pmatrix}
            0 & \mathbb{I}_n \\
            -\mathbb{I}_n & 0
          \end{pmatrix}.
        \end{equation*}

        Estos grupos $\GL_n$, $\mathrm{SL}_n$, $\mathrm{O}_n$, $\mathrm{SO}_n$, $\mathrm{Sp}_n$ forman la clase de los \emph{grupos clĆ”sicos}. NĆ³tese ademĆ”s que todos estos grupos pueden definirse de forma anĆ”loga sobre cualquier cuerpo o incluso sobre $\mathbb{Z}$. \end{equation*}
donde

\end{equation*}
\begin{equation*}

\end{equation*}

\begin{defn}

  \begin{enumerate}

    \item $(R_h)^*\omega = \Ad_{h^{-1}} \omega$, para todo $h\in H$,
    \item $\omega(X^+)=X$, para todo $X\in \fh$.
  \end{enumerate}

  Dada una geometrĆ­a de Cartan $(P,\omega)$ modelada en una geometrĆ­a de Klein $(G,H)$, se define su \emph{curvatura} como la $2$-forma $\fg$-valorada

\end{defn}

\subsection{} Toda geometrĆ­a de Klein $(G,H)$ define una geometrĆ­a de Cartan en $M=G/H$ modelada sobre ella misma, la \emph{geometrĆ­a de Cartan trivial}, tomando $P=G\rightarrow M=G/H$ la proyecciĆ³n natural y $\omega=\theta \in \Omega^1(G;\fg)$ la $1$-forma de Maurer-Cartan. \begin{prop}

\begin{center}

  \end{tikzcd}
\end{center}

\end{prop}

Ahora, como $(P,\omega)$ es plana, la $1$-forma $\omega$ satisface la ecuaciĆ³n de estructura
  \begin{equation*}

  \end{equation*}
  y tambiĆ©n la satisface la $1$-forma $s^* \omega \in \Omega^1(U;\fg)$. Por tanto, reduciendo $U$ si es necesario, existe una Ćŗnica aplicaciĆ³n diferenciable $f:U\rightarrow G$ con $f(x_0)=1$ y tal que $f^*\theta=s^* \omega$. Tomamos ahora una nueva secciĆ³n $s'=s f$, y definimos

  \end{equation*}
  donde $\rho:G\rightarrow G/H$ es la proyecciĆ³n natural. Por construcciĆ³n, el diagrama
\begin{center}
  \begin{tikzcd}
   V \ar{r}{\psi} \ar{d}{\pi} & G \ar{d}{\rho} \\
   U \ar{r}{\varphi} & G/H,
  \end{tikzcd}
\end{center}

\begin{equation*}

\end{equation*}

\begin{pmatrix}

\end{equation*}

\begin{equation*}

\end{equation*}

\end{equation*}
