\begin{defn}
Una \emph{categorÃ­a} $\sfC$ consiste en
\begin{itemize}
  \item una clase $\mathrm{Obj}(\sfC)$, cuyos elementos se llaman los \emph{objetos} de $\sfC$,
\item para cada par de objetos $A$ y $B$ de $\sfC$, una clase $\Hom_{\sfC}(A,B)=\Hom(A,B)$, cuyos elementos $f$ se denominan \emph{morfismos de $A$ a $B$} y se denotan como flechas $f:A\rightarrow B$,
\end{itemize}
y una \emph{ley de composiciÃģn}, para cada terna $(A,B,C)$ de objetos de $\sfC$
\begin{equation*}
\circ_{\sfC}=\circ: \Hom(A,B) \times \Hom(B,C) \rightarrow \Hom(A,C), \ (f,g)\mapsto g\circ f
\end{equation*}
que cumple la propiedad asociativa y tal que, para cada objeto $A$ de $\sfC$, existe un morfismo $\id_A:A\rightarrow A$ que actÃša como la unidad para la operaciÃģn de composiciÃģn.

Si, para cada par de objetos $(A,B)$ de $\sfC$, la clase $\Hom(A,B)$ es un conjunto, entonces decimos que $\sfC$ es una categorÃ­a \emph{localmente pequeÃąa}. Si, ademÃĄs, la clase $\mathrm{Obj}(\sfC)$ es un conjunto, decimos que $\sfC$ es una categorÃ­a \emph{pequeÃąa}.
\end{defn}

\begin{defn}
  Sea $f:A\rightarrow B$ un morfismo en una categorÃ­a $\sfC$. Decimos que $f$ es
  \begin{itemize}
  \item  un \emph{monomorfismo} si, para cualquier otro objeto $C$ de $\sfC$ y cualesquiera dos morfismos $g_1,g_2:C\rightarrow A$, se tiene que si $f\circ g_1 = f\circ g_2$, entonces $g_1=g_2$;
  \item un \emph{epimorfismo} si, para cualquier otro objeto $C$ de $\sfC$ y cualesquiera dos morfismos $g_1,g_2:B\rightarrow C$, se tiene que si $g_1 \circ f=g_2 \circ f$, entonces $g_1=g_2$;
 \item un \emph{isomorfismo} si existe otro morfismo $f^{-1}:B\rightarrow A$ de modo que $f^{-1}\circ f = \id_A$ y $f\circ f^{-1}= \id_B$.
  \end{itemize}
\end{defn}

\begin{rmk}
NÃģtese que un isomorfismo no siempre es lo mismo que un morfismo que es un monomorfismo y un epimorfismo. Por ejemplo, una aplicaciÃģn biyectiva y continua no es necesariamente un homeomorfismo.
\end{rmk}

\begin{defn}
Un \emph{funtor} (o \emph{funtor covariante}) $F:\sfC\rightarrow \sfD$ entre dos categorÃ­as $\sfC$ y $\sfD$, asigna, a cada objeto $A$ de $\sfC$, un objeto $F(A)$ de $\sfD$ y, a cada morfismo $f:A\rightarrow B$ en $\sfC$, un morfismo $F(f):F(A)\rightarrow F(B)$, de forma que $F(f\circ g) = F(f) \circ F(g)$.
\end{defn}

\subsection{} Dada una categorÃ­a $\sfC$, podemos definir su \emph{categorÃ­a opuesta} $\sfC^{\op}$ como la categorÃ­a cuyos objetos son los objetos de $\sfC$ y cuyos morfismos cumplen
\begin{equation*}
\Hom_{\sfC^{\op}}(A,B) = \Hom_{\sfC}(B,A)
\end{equation*}
y de modo que $g \circ_{\sfC^{\op}} f = f\circ_{\sfC} g$.
Un \emph{funtor contravariante} de $\sfC$ a $\sfD$ se define como un funtor $F:\sfC^{\op}\rightarrow \sfD$.

\begin{prop}
Dado un funtor $F:\sfC\rightarrow \sfD$, para todo isomorfismo $f:A\rightarrow B$ en $\sfC$, el morfismo $F(f):F(A)\rightarrow F(B)$ es un isomorfismo en $\sfD$, con $F(f)^{-1}=f^{-1}$.
\end{prop}

\subsection{} Es claro que los funtores se pueden componer, y que en cada categorÃ­a $\sfC$ se puede definir su funtor identidad $\id_{\sfC}:\sfC\rightarrow \sfC$, que envÃ­a cada objeto y cada morfismo en sÃ­ mismo. Esto permitirÃ­a a priori definir una categorÃ­a $\mathsf{Cat}$ cuyos objetos son las categorÃ­as y cuyos morfismos son los funtores entre ellas. \begin{defn}

\begin{defn}

\end{defn}

\begin{prop}
Si $(F,G)$ determina una equivalencia de categorÃ­as, entonces los funtores $F$ y $G$ son plenamente fieles y esencialmente sobreyectivos. RecÃ­procamente, si $F$ es un funtor plenamente fiel y esencialmente sobreyectivo, entonces existe otro funtor $G=F^{-1}$ tal que $(F,F^{-1})$ es una equivalencia de categorÃ­as.
\end{prop}

Otra forma de definir equivalencias de categorÃ­as es en tÃĐrminos de adjunciones.

\begin{defn}
  Una \emph{adjunciÃģn} $G\dashv F$ entre dos categorÃ­as $\sfC$ y $\sfD$ es un par de funtores $F:\sfC\rightarrow \sfD$ y $G:\sfD \rightarrow \sfC$ junto con dos transformaciones naturales $\eta:\id_{\sfC} \rightarrow G\circ F$ y $\varepsilon:F\circ G \rightarrow \id_{\sfD}$, llamadas la \emph{unidad} y la \emph{counidad}, respectivamente, de modo que los siguientes diagramas conmutan

    \begin{tikzcd}
      G \ar{rd}{\id_G} \rar{\eta \circ G} & G\circ F \circ G \dar{G \circ \varepsilon} \\
      & G,
    \end{tikzcd}
    \hspace{1cm}
    \begin{tikzcd}
      F \ar{rd}{\id_F} \rar{F\circ \eta} & F \circ G \circ F \dar{\varepsilon \circ F} \\

    \end{tikzcd}
  \end{center}

\begin{defn}

\subsection{Representabilidad y densidad} Sea $\sfC$ una categorÃ­a y sea $F:\sfC^{\op} \rightarrow \Set$ un prehaz. En general, no todo prehaz es un funtor representable, pero sÃ­ que se puede obtener como un colÃ­mite de funtores representables. Para ello, definimos la \emph{categorÃ­a de pares} $\int F$ cuyos objetos son los pares $(A,x)$ con $A$ un objeto de $\sfC$ y $x \in F(A)$, y tal que un morfismo entre dos pares $(A,x)\rightarrow (B,y)$ estÃĄ dado por un morfismo $f:A\rightarrow B$ tal que $F(f)(y)=x$. \end{equation*}

\begin{thm}[Teorema de densidad]
  \begin{equation*}
F = \colim D_F.
  \end{equation*}
\end{thm}

\begin{proof}
  Basta probar que existe, para todo prehaz $G:\sfC^{\op}\rightarrow \Set$, existe una biyecciÃģn natural

Ãste estÃĄ dado por una colecciÃģn de morfismos $\alpha_{A,x}:\underline{A} \rightarrow G$, para cada par $(A,x)$, de modo que para cada morfismo de pares $f:(A,x)\rightarrow (B,y)$, se tiene que $\alpha_{B,y} \circ f^* = \alpha_{A,x}$. El lema de Yoneda dice que hay una biyecciÃģn natural $G(A)\cong \Hom(\underline{A},G)$. Por tanto, cada morfismo $\alpha_{A,x}$ corresponde a un elemento $g_{A,x} \in G(A)$ de modo que $(Gf)(g_{B,y})= g_{A,x}$. Definimos ahora las aplicaciones

\end{proof}

\subsection{Teorema de representabilidad}
Sea $\Ab$ la categorÃ­a de los grupos abelianos, y sea $F:\sfA^{\op}\rightarrow \Ab$ un funtor exacto por la izquierda. \begin{thm}[Teorema de representabilidad]

\begin{equation*}

\end{equation*}
AdemÃĄs, si $F$ cumple que, para todo sistema directo $I'\rightarrow \sfA$,
\begin{equation*}
F(\varinjlim_{i \in I'} X_i) = \varprojlim_{i\in I'} F(X_i),
\end{equation*}
entonces $F$ es representable; es decir existe un objeto $A$ de $\sfA$ tal que
\begin{equation*}
F= \underline{A} = \Hom(-,A).
\end{equation*}
\end{thm}

\begin{proof}
Para un par minimal $(A,x)$ y cualesquiera dos morfismos de pares $f_1,f_2:(B,y)\rightarrow (A,x)$, se tiene que $f_1=f_2$. En efecto, tenemos una sucesiÃģn exacta en $\sfA$
\begin{center}
  \begin{tikzcd}

  \end{tikzcd}
\end{center}
Invirtiendo el sentido de las flechas y aplicando $F$, obtenemos una sucesiÃģn exacta
\begin{center}
  \begin{tikzcd}

  \end{tikzcd}
\end{center}
Ahora,
\begin{equation*}

\end{equation*}
de modo que tenemos un epimorfismo de pares $p:(A,x) \rightarrow (\coker(f_2-f_1),x)$. Ahora, como $(A,x)$ es minimal, este morfismo es un isomorfismo, y por tanto $\coker(f_2-f_1)=A$, con lo cual $f_1=f_2$. Esto nos permite identificar la subcategorÃ­a de los pares minimales con un conjunto dirigido $I$. Como todo par admite un morfismo a un par minimal, para calcular el colÃ­mite de $D_F:\int F \rightarrow \PSh(C)$, basta hacerlo con los pares minimales, de modo que
\begin{equation*}

\end{equation*}

Finalmente, nÃģtese que la familia $x=(x_i)_{i\in I}$ es un elemento de $\varprojlim_{i\in I} F(A_i)$. Definimos entonces
\begin{equation*}

\end{equation*}
 Si ademÃĄs $F$ envÃ­a lÃ­mites inductivos a lÃ­mites proyectivos, entonces
\begin{equation*}

\end{equation*}
de modo que $x \in F(A)$. Tenemos entonces un par $(A,x)$. Como existe un morfismo $(A,x)\rightarrow (A',x')$ a un par minimal, entonces existe un morfismo de cualquier par al par $(A',x')$ y en particular ha de suceder que $(A',x')=(A,x)$. Concluimos que
\begin{equation*}

\end{equation*}
de modo que $F$ es representable.
\end{proof}

\end{equation*}

\subsection{Soportes compactos}

\begin{defn}
  Sea $X$ una objeto de $\mathsf{Var}$. Para cada $k\in \Z$, se define el \emph{$k$-ÃĐsimo grupo de cohomologÃ­a de $X$} (con coeficientes en $\RR$) como el grupo de cohomologÃ­a de haces

  \end{equation*}

MÃĄs generalmente, si $(X,A)$ es un par formado por un objeto $X$ de $\mathsf{Var}$ y un subespacio $A\subset X$, entonces se define el \emph{$k$-ÃĐsimo grupo de cohomologÃ­a relativa de $(X,A)$} (con coeficientes en $\RR$) como el grupo de cohomologÃ­a de haces
\begin{equation*}

\end{equation*}

\end{defn}

\begin{rmk}
NÃģtese que $H^k(X,\RR)=H^k(X,\varnothing;\RR)$.
\end{rmk}

\subsection{Axiomas de Eilenberg--Steenrod} Los grupos de cohomologÃ­a se pueden definir de manera axiomÃĄtica como una colecciÃģn de funtores contravariantes $H^k$, para $k\in \Z$, de la categorÃ­a de pares $(X,A)$ a la categorÃ­a de grupos abelianos que cumplen las propiedades que enunciamos a continuaciÃģn. De ahora en adelante, si $f$ es una aplicaciÃģn continua entre pares, denotamos $f^*=H^k(f)$ para cualquier $k$.

\subsubsection{Axioma de escisiÃģn} Dado un par $(X,A)$ y un subconjunto $Z\subset A$ cuya adherencia estÃĄ totalmente contenida en el interior de $A$, entonces la inclusiÃģn $i:(X\setminus Z, A\setminus Z) \rightarrow (X,A)$ determina un isomorfismo
\begin{equation*}

\end{equation*}
para cada $k\in \Z$.

\subsubsection{Axioma de dimensiÃģn} $H^0(*)=\RR$ y $H^k(*)=0$ para todo $k\neq 0$.

\subsubsection{Axioma de aditividad} Si $X=\coprod_i X_i$ es una uniÃģn disjunta, entonces $H^k(X,\RR)=\bigoplus_i H^k(X_i,\RR)$.

\subsubsection{Axioma de exactitud} Todo par $(X,A)$ determina una sucesiÃģn exacta larga en cohomologÃ­a
\begin{center}
  \begin{tikzcd}[column sep=1em]
\cdots \rar & H^{k}(X,A;\RR) \rar{j^*} & H^k(X,\RR) \rar{i^*} & H^k(A,\RR) \rar & H^{k+1}(X,A;\RR) \rar & \cdots,
  \end{tikzcd}
\end{center}
donde $i:A\rightarrow X$ y $j:(X,\varnothing)\rightarrow (X,A)$ son las inclusiones naturales.

\subsubsection{Axioma de homotopÃ­a} Si $f,g:(X,A)\rightarrow (Y,B)$ son dos aplicaciones relativamente homotÃģpicas, entonces $f^*=g^*$.

\begin{thm}
Los grupos de cohomologÃ­a que hemos definido cumplen los axiomas de Eilenberg--Steenrod.
\end{thm}

\begin{proof}
  El axioma de dimensiÃģn es inmediato, mientras que el de aditividad se sigue de el hecho de que $\underline{\RR}_X$ es un haz. Para ver la propiedad de escisiÃģn, observamos que si $Z\subset A$ es un subconjunto cuya adherencia estÃĄ contenida en el interior de $A$, entonces $U=X\setminus A = (X\setminus Z)\setminus (A\setminus Z)$, por lo que tenemos un triÃĄngulo de inclusiones
  \begin{center}
    \begin{tikzcd}
      U \ar{rd}{j} \rar{j'} & X\setminus Z \dar{i} \\
       & X.
    \end{tikzcd}
  \end{center}
  Por funtorialidad, tenemos $j_!\underline{\RR}_U=(i_* \circ (j')_!)(\underline{\RR}_U)$, mientras que por la adjunciÃģn $i_*\dashv i^{-1}$ tenemos un isomorfismo
\begin{equation*}

\end{equation*}
El axioma de exactitud se sigue de la sucesiÃģn exacta de haces
\begin{center}
  \begin{tikzcd}

  \end{tikzcd}
\end{center}

\end{proof}

\begin{defn}

\subsection{} NÃģtese la funtorialidad de las construcciones. Para cada aplicaciÃģn diferenciable $f:M\rightarrow N$ tenemos una aplicaciÃģn inducida $f^*:\Omega^\bullet(N)\rightarrow \Omega^\bullet(M)$ que envÃ­a cada forma diferenciable a su pull-back. Puesto que toda aplicaciÃģn continua se puede aproximar por aplicaciones diferenciables, bastarÃĄ considerar homotopÃ­as diferenciables entre aplicaciones diferenciables.

\begin{prop}
  Sean $M$ y $N$ dos variedades diferenciables, $f_0,f_1:M\rightarrow N$ dos aplicaciones diferenciables y $f:M\times [0,1] \rightarrow N$ una homotopÃ­a diferenciable entre $f_0$ y $f_1$. \end{prop}

\begin{proof}
  Basta demostrar que existe una homotopÃ­a $h_0$ entre los morfismos de complejos $i_0^*,i_1^*:\Omega^\bullet(M\times [0,1])\rightarrow \Omega^\bullet(M)$, donde $i_0,i_1:M\rightarrow M\times[0,1]$ son las inclusiones $i_0(x)=(x,0)$ e $i_1(x)=(x,1)$. En efecto, como $f_0=f\circ i_0$ y $f_1=f\circ i_1$, si tomamos $h=h_0 \circ f^*$ tenemos que, para toda forma diferencial $\alpha$,
  \begin{equation*}
h(d\alpha) + d(h\alpha) = h_0(f^* d\alpha) + d(h_0 f^*\alpha) = (h_0d+dh_0)(f^*\alpha)=(i_1^*-i_0^*)(f^*\alpha) = (g^*-f^*)(\alpha).
  \end{equation*}
  Finalmente, la homotopÃ­a $h_0$ puede construirse como la colecciÃģn de aplicaciones
  \begin{align*}
    h_0: \Omega^k(M\times[0,1]) & \longrightarrow \Omega^{k-1}(M) \\
    \alpha  & \longmapsto \int_0^1 i_{\partial_t}\omega dt,
  \end{align*}
  donde $t$ es la coordenada en $[0,1]$. En efecto, para ver que $h_0d+dh_0=i_1^*-i_0^*$ basta calcular en coordenadas locales $x^1,\dots,x^n$. Por la linealidad de $h_0$, basta probarlo en los casos
  \begin{enumerate}
  \item  $\alpha=\alpha(x,t)dt\wedge dx^{i_1}\wedge \cdots \wedge dx^{i_{k-1}}$
  \item $\alpha=\alpha(x,t)dx^{i_1}\wedge\cdots\wedge dx^{i_k}$.
  \end{enumerate}
  Para el caso $1$
  \begin{align*}
  d(h_0\alpha) &= d \left[ \left( \int_0^1 \alpha(t,x) dt \right) dx^{i_1}\wedge \cdots \wedge dx^{i_{k-1}} \right] \\
  &=   \partial_j\left( \int_0^1 \alpha(t,x) dt \right) dx^j\wedge dx^{i_1}\wedge \cdots \wedge dx^{i_{k-1}} \\
  &=  \left( \int_0^1 \partial_j \alpha(t,x) dt \right) dx^j\wedge dx^{i_1}\wedge \cdots \wedge dx^{i_{k-1}},
  \end{align*}
  \begin{align*}
    h_0(d\alpha) &= h \left( \partial_j \alpha(t,x) dt\wedge dx^j\wedge dx^{i_1}\wedge \cdots \wedge dx^{i_{k-1}} \right) \\
    &= \int_0^1 \partial_j \alpha(t,x) i_{\partial_t}(dt\wedge dx^j\wedge dx^{i_1}\wedge \cdots \wedge dx^{i_{k-1}}) dt \\
  &= - \left( \int_0^1 \partial_j \alpha(t,x) dt \right) dx^j\wedge dx^{i_1}\wedge \cdots \wedge dx^{i_{k-1}}.
  \end{align*}
  Luego $dh_0(\alpha)-h_0d(\alpha)=0$. Por otra parte,
  \begin{align*}
(i_0^* dt)_{(x,t)} &= d(t \circ i_0)_{(x,t)} = d(0) = 0 \\
(i_1^* dt)_{(x,t)} &= d(t \circ i_1)_{(x,t)} = d(1) = 0 .
  \end{align*}
  Ahora, para el caso $2$ tenemos que $i_{\partial_t}\alpha = 0$, luego $dh_0\alpha=0$, mientras que
  \begin{equation*}

  \end{equation*}
En ambos casos, concluimos que $h_0$ da una homotopÃ­a entre $i_0^*$ e $i_1^*$.
\end{proof}

\end{lemma}

\subsection{} Como la segunda forma fundamental es simÃĐtrica, el operador de forma $S_p$ es autoadjunto, por lo que es diagonalizable y tiene autovalores reales $\lambda_1,\lambda_2 \in \RR$. Distinguimos dos casos:

\begin{thm}[Gauss--Bonnet]

\subsection{} Recordemos que la caracterÃ­stica de Euler de una superficie $S$ se definÃ­a como \[\chi(S)=C-A+V,\] donde $C$, $A$ y $V$ son el nÃšmero de caras, aristas y vÃĐrtices, respectivamente, de cualquier triangulaciÃģn o, mÃĄs generalmente, de cualquier descomposiciÃģn en celdas de $S$.   \end{equation*}
Basta ver entonces que $\deg \bm{n} = -\chi(S)/2$.

\begin{thm}[Teorema egregio de Gauss]
  Dada una superficie $S\subset E$ y un punto $p\in S$, se tiene

  \end{equation*}

\begin{proof}

  \end{equation*}

  \begin{equation*}

  \end{equation*}

  \end{equation*}

\subsection{} Procedemos ahora a dar una interpretaciÃģn geomÃĐtrica de los sÃ­mbolos de Christoffel. Comenzamos estudiando la nociÃģn de \emph{paralelismo}.
